local oldIndex
oldIndex = hookmetamethod(game, "__index", function(self, key)
    if Settings.Aim.SilentAim and typeof(self) == "Instance" and self:IsA("Mouse") and key == "Hit" then
        local best, bestDist = nil, math.huge
        for _, p in pairs(Players:GetPlayers()) do
            if p ~= LocalPlayer and p.Character then
                local root = p.Character:FindFirstChild("HumanoidRootPart")
                local hum  = p.Character:FindFirstChild("Humanoid")
                if root and hum and hum.Health > 0 then
                    local sp, onScreen = Camera:WorldToViewportPoint(root.Position)
                    if onScreen then
                        local d = (Vector2.new(sp.X, sp.Y) - Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)).Magnitude
                        if d < bestDist then bestDist = d; best = root end
                    end
                end
            end
        end
        if best then return best.CFrame end
    end
    return oldIndex(self, key)
end)

RunService.RenderStepped:Connect(function()
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and p.Character then
            local root = p.Character:FindFirstChild("HumanoidRootPart")
            if root then
                if Settings.Aim.HitboxExpander then
                    local sz = Settings.Aim.HitboxSize
                    root.Size         = Vector3.new(sz, sz, sz)
                    root.Transparency = 0.85
                else
                    root.Size         = Vector3.new(2, 2, 1)
                    root.Transparency = 1
                end
            end
        end
    end
end)

local swordTimer = 0
RunService.Heartbeat:Connect(function(dt)
    if not Settings.Other.SwordAuto then return end
    swordTimer = swordTimer + dt
    if swordTimer < 0.05 then return end
    swordTimer = 0
    local char = LocalPlayer.Character
    if not char then return end
    for _, tool in pairs(char:GetChildren()) do
        if tool:IsA("Tool") then
            pcall(function() tool:Activate() end)
            pcall(function()
                local h = tool:FindFirstChildWhichIsA("Script") or tool:FindFirstChildWhichIsA("LocalScript")
                if h then
                    for _, v in pairs(tool:GetDescendants()) do
                        if v:IsA("RemoteEvent") then
                            pcall(function() v:FireServer() end)
                        end
                    end
                end
            end)
            pcall(function()
                for _, v in pairs(tool:GetDescendants()) do
                    if v:IsA("RemoteEvent") then
                        pcall(function() v:FireServer() end)
                    end
                end
            end)
        end
    end
    pcall(function() mouse1press() end)
    pcall(function()
        task.delay(0.02, function()
            pcall(function() mouse1release() end)
        end)
    end)
end)

local attractBVs = {}
RunService.Heartbeat:Connect(function(dt)
    if not Settings.Other.AttractPlayers then
        for p, bv in pairs(attractBVs) do
            pcall(function() bv:Destroy() end)
            attractBVs[p] = nil
        end
        return
    end
    local myChar = LocalPlayer.Character
    local myRoot = myChar and myChar:FindFirstChild("HumanoidRootPart")
    if not myRoot then return end
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and p.Character then
            local root = p.Character:FindFirstChild("HumanoidRootPart")
            local hum  = p.Character:FindFirstChildWhichIsA("Humanoid")
            if root and hum and hum.Health > 0 then
                local diff = myRoot.Position - root.Position
                local dist = diff.Magnitude
                if dist > 2 then
                    local dir   = diff.Unit
                    local speed = math.clamp(dist * 10, 30, 150)
                    pcall(function()
                        local bv = root:FindFirstChild("_ABV")
                        if not bv then
                            bv          = Instance.new("BodyVelocity")
                            bv.Name     = "_ABV"
                            bv.MaxForce = Vector3.new(1e9, 1e9, 1e9)
                            bv.P        = 1e9
                            bv.Parent   = root
                            attractBVs[p] = bv
                        end
                        bv.Velocity = dir * speed
                    end)
                    pcall(function()
                        root.CFrame = root.CFrame:Lerp(
                            CFrame.new(myRoot.Position - dir * 1.5),
                            math.min(dt * 8, 1)
                        )
                    end)
                else
                    pcall(function()
                        if attractBVs[p] then
                            attractBVs[p].Velocity = Vector3.new(0, 0, 0)
                        end
                    end)
                end
            end
        end
    end
end)

Players.PlayerRemoving:Connect(function(p)
    pcall(function()
        if attractBVs[p] then
            attractBVs[p]:Destroy()
            attractBVs[p] = nil
        end
    end)
end)

local espConns = {}
local espDraw  = {}

local function DestroyESP(player)
    if espConns[player] then
        espConns[player]:Disconnect()
        espConns[player] = nil
    end
    if espDraw[player] then
        for _, d in pairs(espDraw[player].esp) do pcall(function() d:Remove() end) end
        for _, d in pairs(espDraw[player].limbs) do pcall(function() d:Remove() end) end
        espDraw[player] = nil
    end
end

local function CreateFullESP(player)
    DestroyESP(player)
    local esp = {
        box       = Drawing.new("Square"),
        line      = Drawing.new("Line"),
        healthBar = Drawing.new("Line"),
        name      = Drawing.new("Text"),
    }
    local limbs = {}
    for i = 1, 14 do limbs[i] = Drawing.new("Line") end
    espDraw[player] = { esp = esp, limbs = limbs }
    esp.box.Filled          = false
    esp.name.Size           = 12
    esp.name.Center         = true
    esp.name.Outline        = true
    esp.healthBar.Thickness = 1.5

    local function drawBone(line, p1, p2)
        if p1 and p2 and p1 ~= p2 then
            local a = Camera:WorldToViewportPoint(p1.Position)
            local b = Camera:WorldToViewportPoint(p2.Position)
            if a.Z > 0 and b.Z > 0 then
                line.From      = Vector2.new(a.X, a.Y)
                line.To        = Vector2.new(b.X, b.Y)
                line.Color     = currentAutoColor
                line.Thickness = 0.5
                line.Visible   = true
                return
            end
        end
        line.Visible = false
    end

    local function HideAll()
        for _, v in pairs(esp) do pcall(function() v.Visible = false end) end
        for i = 1, #limbs do pcall(function() limbs[i].Visible = false end) end
    end

    local function Update()
        if not espDraw[player] then return end
        local ok = pcall(function()
            local char = player.Character
            local hum  = char and char:FindFirstChildWhichIsA("Humanoid")
            if not char or not char:FindFirstChild("HumanoidRootPart") or not hum or hum.Health <= 0 then
                HideAll(); return
            end
            local root          = char.HumanoidRootPart
            local pos, onScreen = Camera:WorldToViewportPoint(root.Position)
            local distance      = (Camera.CFrame.Position - root.Position).Magnitude
            if onScreen and Settings.ESP.Enabled then
                local sizeX = math.clamp(1000 / distance, 10, 150)
                local sizeY = sizeX * 1.6
                if Settings.ESP.Box then
                    esp.box.Visible   = true
                    esp.box.Size      = Vector2.new(sizeX, sizeY)
                    esp.box.Position  = Vector2.new(pos.X - sizeX / 2, pos.Y - sizeY / 2)
                    esp.box.Color     = currentAutoColor
                    esp.box.Thickness = Settings.ESP.Thickness
                else esp.box.Visible = false end
                if Settings.ESP.Line then
                    esp.line.Visible = true
                    if Settings.ESP.LinePos == "CENTER" then
                        esp.line.From = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
                    elseif Settings.ESP.LinePos == "BOTTOM" then
                        esp.line.From = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y)
                    else
                        esp.line.From = Vector2.new(Camera.ViewportSize.X / 2, 0)
                    end
                    esp.line.To           = Vector2.new(pos.X, pos.Y - sizeY / 2)
                    esp.line.Color        = currentAutoColor
                    esp.line.Thickness    = Settings.ESP.Thickness
                    esp.line.Transparency = math.clamp(1 - (distance / 1200), 0.35, 0.95)
                else esp.line.Visible = false end
                if Settings.ESP.Health then
                    local hp = math.clamp(hum.Health / math.max(hum.MaxHealth, 1), 0, 1)
                    esp.healthBar.Visible = true
                    esp.healthBar.From    = Vector2.new(pos.X - sizeX / 2 - 5, pos.Y + sizeY / 2)
                    esp.healthBar.To      = Vector2.new(pos.X - sizeX / 2 - 5, pos.Y + sizeY / 2 - sizeY * hp)
                    esp.healthBar.Color   = Color3.fromRGB(math.floor(255 - 255 * hp), math.floor(255 * hp), 0)
                else esp.healthBar.Visible = false end
                if Settings.ESP.Name then
                    esp.name.Visible  = true
                    esp.name.Text     = player.Name .. " [" .. math.floor(distance) .. "m]"
                    esp.name.Position = Vector2.new(pos.X, pos.Y - sizeY / 2 - 16)
                    esp.name.Color    = currentAutoColor
                else esp.name.Visible = false end
                if Settings.ESP.Skeleton then
                    local isR15 = char:FindFirstChild("UpperTorso") ~= nil
                    local bones = {}
                    if isR15 then
                        bones = {
                            {char:FindFirstChild("Head"),         char:FindFirstChild("UpperTorso")},
                            {char:FindFirstChild("UpperTorso"),   char:FindFirstChild("LowerTorso")},
                            {char:FindFirstChild("UpperTorso"),   char:FindFirstChild("LeftUpperArm")},
                            {char:FindFirstChild("LeftUpperArm"), char:FindFirstChild("LeftLowerArm")},
                            {char:FindFirstChild("LeftLowerArm"), char:FindFirstChild("LeftHand")},
                            {char:FindFirstChild("UpperTorso"),   char:FindFirstChild("RightUpperArm")},
                            {char:FindFirstChild("RightUpperArm"),char:FindFirstChild("RightLowerArm")},
                            {char:FindFirstChild("RightLowerArm"),char:FindFirstChild("RightHand")},
                            {char:FindFirstChild("LowerTorso"),   char:FindFirstChild("LeftUpperLeg")},
                            {char:FindFirstChild("LeftUpperLeg"), char:FindFirstChild("LeftLowerLeg")},
                            {char:FindFirstChild("LeftLowerLeg"), char:FindFirstChild("LeftFoot")},
                            {char:FindFirstChild("LowerTorso"),   char:FindFirstChild("RightUpperLeg")},
                            {char:FindFirstChild("RightUpperLeg"),char:FindFirstChild("RightLowerLeg")},
                            {char:FindFirstChild("RightLowerLeg"),char:FindFirstChild("RightFoot")},
                        }
                    else
                        bones = {
                            {char:FindFirstChild("Head"),     char:FindFirstChild("Torso")},
                            {char:FindFirstChild("Torso"),    char:FindFirstChild("Left Arm")},
                            {char:FindFirstChild("Torso"),    char:FindFirstChild("Right Arm")},
                            {char:FindFirstChild("Torso"),    char:FindFirstChild("Left Leg")},
                            {char:FindFirstChild("Torso"),    char:FindFirstChild("Right Leg")},
                        }
                    end
                    for i = 1, #limbs do
                        drawBone(limbs[i], bones[i] and bones[i][1], bones[i] and bones[i][2])
                    end
                else
                    for i = 1, #limbs do limbs[i].Visible = false end
                end
            else
                HideAll()
            end
        end)
        if not ok then HideAll() end
    end

    espConns[player] = RunService.RenderStepped:Connect(Update)
    player.CharacterAdded:Connect(function()
        task.wait(1)
        if player and player.Parent then CreateFullESP(player) end
    end)
end

local TabBtns = {}
local function SwitchTab(name)
    Settings.CurrentTab = name
    for n, p in pairs(Pages) do p.Visible = (n == name) end
    for n, b in pairs(TabBtns) do
        if n == name then
            b.BackgroundColor3 = Color3.fromRGB(0, 80, 200)
            b.TextColor3       = Color3.fromRGB(255, 255, 255)
            local s = b:FindFirstChildOfClass("UIStroke")
            if s then s.Color = Color3.fromRGB(0, 140, 255) end
        else
            b.BackgroundColor3 = Color3.fromRGB(10, 12, 24)
            b.TextColor3       = Color3.fromRGB(110, 115, 145)
            local s = b:FindFirstChildOfClass("UIStroke")
            if s then s.Color = Color3.fromRGB(20, 22, 40) end
        end
    end
end

        local on = tbl[key]
        Track.BackgroundColor3  = on and Color3.fromRGB(0, 100, 220) or Color3.fromRGB(22, 24, 40)
        Knob.Position           = on and UDim2.new(1, -15, 0.5, -6) or UDim2.new(0, 2, 0.5, -6)
        accent.BackgroundColor3 = on and Color3.fromRGB(0, 180, 255) or Color3.fromRGB(40, 44, 70)
    end
    Track.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
            tbl[key] = not tbl[key]; Refresh()
        end
    end)
    Refresh()
end

    local f = Card(page, order, 46)

    LT.Text              = label .. ":  " .. tbl[key]
    LT.Size              = UDim2.new(1, -12, 0, 20)
    LT.Position          = UDim2.new(0, 12, 0, 4)
    LT.Font              = Enum.Font.GothamBold
    LT.TextSize          = 10
    LT.TextColor3        = Color3.fromRGB(200, 205, 225)
    LT.BackgroundTransparency = 1
    LT.TextXAlignment    = Enum.TextXAlignment.Left
    LT.ZIndex            = 5
    local TrackBg = Instance.new("Frame", f)
    TrackBg.Size             = UDim2.new(1, -18, 0, 6)
    TrackBg.Position         = UDim2.new(0, 9, 0, 32)
    TrackBg.BackgroundColor3 = Color3.fromRGB(18, 20, 38)
    TrackBg.ZIndex           = 5
    Instance.new("UICorner", TrackBg).CornerRadius = UDim.new(1, 0)
    local Fill = Instance.new("Frame", TrackBg)
    local pi = (tbl[key] - minV) / (maxV - minV)
    Fill.Size             = UDim2.new(pi, 0, 1, 0)
    Fill.BackgroundColor3 = Color3.fromRGB(0, 140, 255)
    Fill.ZIndex           = 6
    Instance.new("UICorner", Fill).CornerRadius = UDim.new(1, 0)
    local FG = Instance.new("UIGradient", Fill)
    FG.Color    = ColorSequence.new({ ColorSequenceKeypoint.new(0, Color3.fromRGB(0, 180, 255)), ColorSequenceKeypoint.new(1, Color3.fromRGB(80, 0, 255)) })
    FG.Rotation = 0
    local Knob = Instance.new("Frame", TrackBg)
    Knob.Size             = UDim2.new(0, 11, 0, 11)
    Knob.AnchorPoint      = Vector2.new(0.5, 0.5)
    Knob.Position         = UDim2.new(pi, 0, 0.5, 0)
    Knob.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    Knob.ZIndex           = 7
    Instance.new("UICorner", Knob).CornerRadius = UDim.new(1, 0)
    local function Set(ip)
        local pct = math.clamp((ip.X - TrackBg.AbsolutePosition.X) / TrackBg.AbsoluteSize.X, 0, 1)
        local val = minV + pct * (maxV - minV)
        val = maxV <= 10 and (math.floor(val * 10) / 10) or math.floor(val)
        tbl[key] = val; LT.Text = label .. ":  " .. val
        Fill.Size = UDim2.new(pct, 0, 1, 0); Knob.Position = UDim2.new(pct, 0, 0.5, 0)
    end
    local drag = false
    TrackBg.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
            drag = true; Set(i.Position)
        end
    end)
    TrackBg.InputEnded:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
            drag = false
        end
    end)
    UserInputService.InputChanged:Connect(function(i)
        if drag and (i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch) then
            Set(i.Position)
        end
    end)
end

-- ตรวจว่าเป้าหมายไม่ถูกบัง (ไม่มีกำแพงขวาง)
local function IsVisible(targetChar, part)
    if not (targetChar and part) then return false end
    local myChar  = LocalPlayer.Character
    local origin  = Camera.CFrame.Position
    local dir     = part.Position - origin
    local params  = RaycastParams.new()
    params.FilterType = Enum.RaycastFilterType.Exclude
    params.FilterDescendantsInstances = {
        myChar or workspace,
        targetChar
    }
    local result = workspace:Raycast(origin, dir, params)
    return result == nil   -- nil = ไม่มีอะไรขวาง = มองเห็น
end

-- หาผู้เล่นที่ใกล้กลางจอมากที่สุดใน FOV และมองเห็นได้จริง
local function GetClosestPlayer()
    local closest  = nil
    local bestDist = Settings.Aimbot.FOV
    local center   = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
    for _, player in pairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character then
            local char = player.Character
            local hum  = char:FindFirstChild("Humanoid")
            if not (hum and hum.Health > 0) then continue end
            local part = char:FindFirstChild(Settings.Aimbot.TargetPart)
                      or char:FindFirstChild("HumanoidRootPart")
            if not part then continue end
            local pos, onScreen = Camera:WorldToViewportPoint(part.Position)
            if not onScreen then continue end
            local d = (Vector2.new(pos.X, pos.Y) - center).Magnitude
            if d < bestDist and IsVisible(char, part) then
                closest  = player
                bestDist = d
            end
        end
    end
    return closest
end

-- Aimbot lock: priority 2000 รันหลัง game camera (200) ทุก frame
-- เช็ค FOV + visibility ทุก frame — ถ้าอยู่หลังกำแพงหรือออกนอก FOV หยุดล็อก
RunService:BindToRenderStep("__AimbotLock", 2000, function()
    if not (Settings.Aimbot.Enabled and (isShooting or isScoping)) then return end
    if not (lockedTarget and lockedTarget.Character) then return end
    local char = lockedTarget.Character
    local part = char:FindFirstChild(Settings.Aimbot.TargetPart)
             or  char:FindFirstChild("HumanoidRootPart")
    if not (part and part.Parent) then return end
    -- เช็ค FOV: เป้าต้องอยู่ในวง FOV
    local sv = Camera:WorldToViewportPoint(part.Position)
    local center = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)
    if sv.Z <= 0 then return end   -- อยู่หลังกล้อง
    if (Vector2.new(sv.X, sv.Y) - center).Magnitude > Settings.Aimbot.FOV then return end
    -- เช็คกำแพง: ถ้ามีอะไรขวางให้หยุดล็อก frame นั้น
    if not IsVisible(char, part) then return end
    Camera.CFrame = CFrame.new(Camera.CFrame.Position, part.Position)
end)

-- [[ 3. ESP ENGINE ]]
local function CreateFullESP(player)
    local esp = {
        box       = Drawing.new("Square"),
        line      = Drawing.new("Line"),
        healthBar = Drawing.new("Line"),
        name      = Drawing.new("Text"),
    }
    -- 14 segments รองรับทั้ง R6 และ R15 (R6 ใช้แค่ 5, ที่เหลือ hidden อัตโนมัติ)
    local limbs = {}
    for i = 1, 14 do limbs[i] = Drawing.new("Line") end
    esp.box.Filled          = false
    esp.name.Size           = 12
    esp.name.Center         = true
    esp.name.Outline        = true
    esp.healthBar.Thickness = 1.5

    -- วาดเส้นระหว่าง 2 parts — เช็คแค่ Z > 0 (อยู่หน้ากล้อง)
    -- ไม่บังคับ onScreen เพราะถ้า endpoint อยู่นอก viewport นิดหน่อยเส้นก็ควรติด
    local function drawBone(line, p1, p2)
        if p1 and p2 and p1 ~= p2 and Settings.ESP.Enabled and Settings.ESP.Skeleton then
            local a = Camera:WorldToViewportPoint(p1.Position)
            local b = Camera:WorldToViewportPoint(p2.Position)
            if a.Z > 0 and b.Z > 0 then
                line.From = Vector2.new(a.X, a.Y); line.To = Vector2.new(b.X, b.Y)
                line.Color = currentAutoColor; line.Thickness = 0.5; line.Visible = true
                return
            end
        end
        line.Visible = false
    end

    local function Update()
        local char = player.Character
        local hum  = char and char:FindFirstChild("Humanoid")
        if not char or not char:FindFirstChild("HumanoidRootPart")
           or not hum or hum.Health <= 0 then
            for _, v in pairs(esp)   do v.Visible = false end
            for i = 1, #limbs do limbs[i].Visible = false end
            return
        end

        local root = char.HumanoidRootPart
        local pos, onScreen = Camera:WorldToViewportPoint(root.Position)
        local distance = (Camera.CFrame.Position - root.Position).Magnitude

        -- Box / Line / Health / Name: ต้องการ root on screen
        if onScreen and Settings.ESP.Enabled then
            local sizeX = math.clamp(1000 / distance, 10, 150)
            local sizeY = sizeX * 1.6

            if Settings.ESP.Box then
                esp.box.Visible  = true
                esp.box.Size     = Vector2.new(sizeX, sizeY)
                esp.box.Position = Vector2.new(pos.X - sizeX/2, pos.Y - sizeY/2)
                esp.box.Color    = currentAutoColor; esp.box.Thickness = 0.5
            else esp.box.Visible = false end

            if Settings.ESP.Line then
                esp.line.Visible = true
                if Settings.ESP.LinePos == "CENTER" then
                    esp.line.From = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)
                elseif Settings.ESP.LinePos == "BOTTOM" then
                    esp.line.From = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y)
                else
                    esp.line.From = Vector2.new(Camera.ViewportSize.X/2, 0)
                end
                esp.line.To           = Vector2.new(pos.X, pos.Y - sizeY/2)
                esp.line.Color        = currentAutoColor; esp.line.Thickness = 0.5
                esp.line.Transparency = math.clamp(1 - (distance/1200), 0.35, 0.95)
            else esp.line.Visible = false end

            if Settings.ESP.Health then
                local hp = hum.Health / hum.MaxHealth
                esp.healthBar.Visible = true
                esp.healthBar.From  = Vector2.new(pos.X - sizeX/2 - 5, pos.Y + sizeY/2)
                esp.healthBar.To    = Vector2.new(pos.X - sizeX/2 - 5, pos.Y + sizeY/2 - sizeY * math.clamp(hp,0,1))
                esp.healthBar.Color = Color3.fromRGB(math.floor(255-255*hp), math.floor(255*hp), 0)
            else esp.healthBar.Visible = false end

            if Settings.ESP.Name then
                esp.name.Visible  = true
                esp.name.Text     = player.Name .. " [" .. math.floor(distance) .. "m]"
                esp.name.Position = Vector2.new(pos.X, pos.Y - sizeY/2 - 16)
                esp.name.Color    = currentAutoColor
            else esp.name.Visible = false end
        else
            esp.box.Visible = false; esp.line.Visible = false
            esp.healthBar.Visible = false; esp.name.Visible = false
        end

        -- Skeleton: แต่ละ segment เช็คตัวเองอิสระจาก root onScreen
        local isR15  = char:FindFirstChild("UpperTorso") ~= nil
        local bones  = {}   -- {p1, p2} pairs
        if isR15 then
            local head    = char:FindFirstChild("Head")
            local uTorso  = char:FindFirstChild("UpperTorso")
            local lTorso  = char:FindFirstChild("LowerTorso")
            local lUA     = char:FindFirstChild("LeftUpperArm")
            local lLA     = char:FindFirstChild("LeftLowerArm")
            local lHand   = char:FindFirstChild("LeftHand")
            local rUA     = char:FindFirstChild("RightUpperArm")
            local rLA     = char:FindFirstChild("RightLowerArm")
            local rHand   = char:FindFirstChild("RightHand")
            local lUL     = char:FindFirstChild("LeftUpperLeg")
            local lLL     = char:FindFirstChild("LeftLowerLeg")
            local lFoot   = char:FindFirstChild("LeftFoot")
            local rUL     = char:FindFirstChild("RightUpperLeg")
            local rLL     = char:FindFirstChild("RightLowerLeg")
            local rFoot   = char:FindFirstChild("RightFoot")
            bones = {
                {head,   uTorso}, {uTorso, lTorso},
                {uTorso, lUA},    {lUA,    lLA},    {lLA, lHand},
                {uTorso, rUA},    {rUA,    rLA},    {rLA, rHand},
                {lTorso, lUL},    {lUL,    lLL},    {lLL, lFoot},
                {lTorso, rUL},    {rUL,    rLL},    {rLL, rFoot},
            }
        else  -- R6
            local head  = char:FindFirstChild("Head")
            local torso = char:FindFirstChild("Torso")
            local lArm  = char:FindFirstChild("Left Arm")
            local rArm  = char:FindFirstChild("Right Arm")
            local lLeg  = char:FindFirstChild("Left Leg")
            local rLeg  = char:FindFirstChild("Right Leg")
            bones = {
                {head,  torso},
                {torso, lArm}, {torso, rArm},
                {torso, lLeg}, {torso, rLeg},
            }
        end
        for i = 1, #limbs do
            drawBone(limbs[i], bones[i] and bones[i][1], bones[i] and bones[i][2])
        end
    end
    RunService.RenderStepped:Connect(Update)
end

for _, p in pairs(Players:GetPlayers()) do
    if p ~= LocalPlayer then CreateFullESP(p) end
end
Players.PlayerAdded:Connect(CreateFullESP)

    Knob.Size = UDim2.new(0, 12, 0, 12); Knob.AnchorPoint = Vector2.new(0.5, 0.5)
    Knob.Position = UDim2.new((targetTable[key]-minVal)/(maxVal-minVal), 0, 0.5, 0)
    Knob.BackgroundColor3 = Color3.fromRGB(255, 255, 255); Knob.ZIndex = 7
    Instance.new("UICorner", Knob).CornerRadius = UDim.new(0, 6)

    local function Set(ip)
        local pct = math.clamp((ip.X - Track.AbsolutePosition.X) / Track.AbsoluteSize.X, 0, 1)
        local val = minVal + pct * (maxVal - minVal)
        val = maxVal <= 10 and (math.floor(val*10)/10) or math.floor(val)
        targetTable[key] = val; LT.Text = label .. ":  " .. val
        Fill.Size = UDim2.new(pct, 0, 1, 0); Knob.Position = UDim2.new(pct, 0, 0.5, 0)
        Fill.BackgroundColor3 = currentAutoColor
    end
    local drag = false
    Bg.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
            drag = true; Set(i.Position)
        end
    end)
    Bg.InputEnded:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
            drag = false
        end
    end)
    UserInputService.InputChanged:Connect(function(i)
        if drag and (i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch) then
            Set(i.Position)
        end
    end)
end

    local TL = Instance.new("TextLabel", Frame)
    TL.Text = name; TL.Size = UDim2.new(0.5, 0, 1, 0); TL.Position = UDim2.new(0, 12, 0, 0)
    TL.Font = Enum.Font.GothamBold; TL.TextSize = 12; TL.TextColor3 = Color3.fromRGB(220, 220, 235)
    TL.BackgroundTransparency = 1; TL.TextXAlignment = Enum.TextXAlignment.Left; TL.ZIndex = 5

    local Btn = Instance.new("TextButton", Frame)
    Btn.Font = Enum.Font.GothamBold; Btn.TextSize = 11; Btn.TextColor3 = Color3.fromRGB(0, 200, 255)
    Btn.BackgroundColor3 = Color3.fromRGB(16, 19, 34); Btn.Size = UDim2.new(0, 108, 0, 26)
    Btn.Position = UDim2.new(1, -114, 0.5, -13); Btn.ZIndex = 5
    Instance.new("UICorner", Btn).CornerRadius = UDim.new(0, 6)
    local BS = Instance.new("UIStroke", Btn); BS.Color = Color3.fromRGB(35, 40, 65); BS.Thickness = 1

    local idx = 1
    for i, v in ipairs(options) do if targetTable[key] == v then idx = i; break end end

    Btn.MouseButton1Click:Connect(function() idx = (idx % #options) + 1; Upd() end)
    Upd()
end

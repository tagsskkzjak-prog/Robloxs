    ESP     = { Enabled = true, Skeleton = true, Box = true, Line = true, Name = true, Health = true, Thickness = 0.6, LinePos = "TOP" },
    Sdamage = { Enabled = false, Amount = 100 },
    CurrentTab = "ESP"
}

local RunService       = game:GetService("RunService")
local Players          = game:GetService("Players")
local UserInputService = game:GetService("UserInputService")
local LocalPlayer      = Players.LocalPlayer
local Camera           = game.Workspace.CurrentCamera

local AutoColors = {
    Color3.fromRGB(255, 255, 255),
    Color3.fromRGB(0,   120, 255),
    Color3.fromRGB(255, 220, 0),
    Color3.fromRGB(255, 40,  40),
    Color3.fromRGB(180, 0,   255),
    Color3.fromRGB(0,   255, 220),
}
local autoColorIndex   = 1
local autoColorTimer   = 0
local currentAutoColor = AutoColors[1]

RunService.RenderStepped:Connect(function(dt)
    autoColorTimer = autoColorTimer + dt
    if autoColorTimer >= 1.2 then
        autoColorTimer   = 0
        autoColorIndex   = (autoColorIndex % #AutoColors) + 1
        currentAutoColor = AutoColors[autoColorIndex]
    end
end)

RunService.RenderStepped:Connect(function()
    if Settings.Sdamage.Enabled then
        for _, player in pairs(Players:GetPlayers()) do
            if player ~= LocalPlayer and player.Character then
                local hum  = player.Character:FindFirstChild("Humanoid")
                local root = player.Character:FindFirstChild("HumanoidRootPart")
                if hum and root and hum.Health > 0 then
                    local _, onScreen = Camera:WorldToViewportPoint(root.Position)
                    if onScreen then
                        hum:TakeDamage(Settings.Sdamage.Amount)
                    end
                end
            end
        end
    end
end)

local function CreateFullESP(player)
    local esp = {
        box       = Drawing.new("Square"),
        line      = Drawing.new("Line"),
        healthBar = Drawing.new("Line"),
        name      = Drawing.new("Text"),
    }
    local limbs = {}
    for i = 1, 14 do limbs[i] = Drawing.new("Line") end
    esp.box.Filled          = false
    esp.name.Size           = 12
    esp.name.Center         = true
    esp.name.Outline        = true
    esp.healthBar.Thickness = 1.5

    local function drawBone(line, p1, p2)
        if p1 and p2 and p1 ~= p2 and Settings.ESP.Enabled and Settings.ESP.Skeleton then
            local a = Camera:WorldToViewportPoint(p1.Position)
            local b = Camera:WorldToViewportPoint(p2.Position)
            if a.Z > 0 and b.Z > 0 then
                line.From      = Vector2.new(a.X, a.Y)
                line.To        = Vector2.new(b.X, b.Y)
                line.Color     = currentAutoColor
                line.Thickness = 0.5
                line.Visible   = true
                return
            end
        end
        line.Visible = false
    end

    local function Update()
        local char = player.Character
        local hum  = char and char:FindFirstChild("Humanoid")
        if not char or not char:FindFirstChild("HumanoidRootPart") or not hum or hum.Health <= 0 then
            for _, v in pairs(esp)   do v.Visible = false end
            for i = 1, #limbs do limbs[i].Visible = false end
            return
        end

        local root              = char.HumanoidRootPart
        local pos, onScreen     = Camera:WorldToViewportPoint(root.Position)
        local distance          = (Camera.CFrame.Position - root.Position).Magnitude

        if onScreen and Settings.ESP.Enabled then
            local sizeX = math.clamp(1000 / distance, 10, 150)
            local sizeY = sizeX * 1.6

            if Settings.ESP.Box then
                esp.box.Visible   = true
                esp.box.Size      = Vector2.new(sizeX, sizeY)
                esp.box.Position  = Vector2.new(pos.X - sizeX/2, pos.Y - sizeY/2)
                esp.box.Color     = currentAutoColor
                esp.box.Thickness = 0.5
            else esp.box.Visible = false end

            if Settings.ESP.Line then
                esp.line.Visible = true
                if Settings.ESP.LinePos == "CENTER" then
                    esp.line.From = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)
                elseif Settings.ESP.LinePos == "BOTTOM" then
                    esp.line.From = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y)
                else
                    esp.line.From = Vector2.new(Camera.ViewportSize.X/2, 0)
                end
                esp.line.To           = Vector2.new(pos.X, pos.Y - sizeY/2)
                esp.line.Color        = currentAutoColor
                esp.line.Thickness    = 0.5
                esp.line.Transparency = math.clamp(1 - (distance/1200), 0.35, 0.95)
            else esp.line.Visible = false end

            if Settings.ESP.Health then
                local hp = hum.Health / hum.MaxHealth
                esp.healthBar.Visible = true
                esp.healthBar.From  = Vector2.new(pos.X - sizeX/2 - 5, pos.Y + sizeY/2)
                esp.healthBar.To    = Vector2.new(pos.X - sizeX/2 - 5, pos.Y + sizeY/2 - sizeY * math.clamp(hp, 0, 1))
                esp.healthBar.Color = Color3.fromRGB(math.floor(255 - 255*hp), math.floor(255*hp), 0)
            else esp.healthBar.Visible = false end

            if Settings.ESP.Name then
                esp.name.Visible  = true
                esp.name.Text     = player.Name .. " [" .. math.floor(distance) .. "m]"
                esp.name.Position = Vector2.new(pos.X, pos.Y - sizeY/2 - 16)
                esp.name.Color    = currentAutoColor
            else esp.name.Visible = false end
        else
            esp.box.Visible       = false
            esp.line.Visible      = false
            esp.healthBar.Visible = false
            esp.name.Visible      = false
        end

        local isR15 = char:FindFirstChild("UpperTorso") ~= nil
        local bones = {}
        if isR15 then
            local head   = char:FindFirstChild("Head")
            local uTorso = char:FindFirstChild("UpperTorso")
            local lTorso = char:FindFirstChild("LowerTorso")
            local lUA    = char:FindFirstChild("LeftUpperArm")
            local lLA    = char:FindFirstChild("LeftLowerArm")
            local lHand  = char:FindFirstChild("LeftHand")
            local rUA    = char:FindFirstChild("RightUpperArm")
            local rLA    = char:FindFirstChild("RightLowerArm")
            local rHand  = char:FindFirstChild("RightHand")
            local lUL    = char:FindFirstChild("LeftUpperLeg")
            local lLL    = char:FindFirstChild("LeftLowerLeg")
            local lFoot  = char:FindFirstChild("LeftFoot")
            local rUL    = char:FindFirstChild("RightUpperLeg")
            local rLL    = char:FindFirstChild("RightLowerLeg")
            local rFoot  = char:FindFirstChild("RightFoot")
            bones = {
                {head,   uTorso}, {uTorso, lTorso},
                {uTorso, lUA},    {lUA,    lLA},    {lLA, lHand},
                {uTorso, rUA},    {rUA,    rLA},    {rLA, rHand},
                {lTorso, lUL},    {lUL,    lLL},    {lLL, lFoot},
                {lTorso, rUL},    {rUL,    rLL},    {rLL, rFoot},
            }
        else
            local head  = char:FindFirstChild("Head")
            local torso = char:FindFirstChild("Torso")
            local lArm  = char:FindFirstChild("Left Arm")
            local rArm  = char:FindFirstChild("Right Arm")
            local lLeg  = char:FindFirstChild("Left Leg")
            local rLeg  = char:FindFirstChild("Right Leg")
            bones = {
                {head,  torso},
                {torso, lArm}, {torso, rArm},
                {torso, lLeg}, {torso, rLeg},
            }
        end
        for i = 1, #limbs do
            drawBone(limbs[i], bones[i] and bones[i][1], bones[i] and bones[i][2])
        end
    end
    RunService.RenderStepped:Connect(Update)
end

for _, p in pairs(Players:GetPlayers()) do
    if p ~= LocalPlayer then CreateFullESP(p) end
end

local TabButtons = {}
local function SwitchTab(tabName)
    Settings.CurrentTab = tabName
    for name, page in pairs(Pages) do
        page.Visible = (name == tabName)
    end
    for name, btn in pairs(TabButtons) do
        if name == tabName then
            btn.BackgroundColor3 = Color3.fromRGB(0, 90, 210)
            btn.TextColor3       = Color3.fromRGB(255, 255, 255)
        else
            btn.BackgroundColor3 = Color3.fromRGB(14, 16, 28)
            btn.TextColor3       = Color3.fromRGB(150, 150, 170)
        end
    end
end

local function AddSlider(page, label, minVal, maxVal, targetTable, key)
    local Bg = Instance.new("Frame", page)
    Bg.Size                   = UDim2.new(1, -6, 0, 50)
    Bg.BackgroundColor3       = Color3.fromRGB(10, 11, 20)
    Bg.BackgroundTransparency = 0.05
    Bg.ZIndex                 = 4
    Instance.new("UICorner", Bg).CornerRadius = UDim.new(0, 8)
    local BS = Instance.new("UIStroke", Bg)
    BS.Color     = Color3.fromRGB(22, 26, 44)
    BS.Thickness = 1

    local LT = Instance.new("TextLabel", Bg)
    LT.Text             = label .. ":  " .. targetTable[key]
    LT.Size             = UDim2.new(1, 0, 0, 20)
    LT.Position         = UDim2.new(0, 12, 0, 4)
    LT.Font             = Enum.Font.GothamMedium
    LT.TextColor3       = Color3.fromRGB(0, 200, 255)
    LT.TextSize         = 11
    LT.BackgroundTransparency = 1
    LT.TextXAlignment   = Enum.TextXAlignment.Left
    LT.ZIndex           = 5

    local Track = Instance.new("Frame", Bg)
    Track.Size             = UDim2.new(1, -24, 0, 3)
    Track.Position         = UDim2.new(0, 12, 0, 34)
    Track.BackgroundColor3 = Color3.fromRGB(22, 26, 44)
    Track.ZIndex           = 5
    Instance.new("UICorner", Track).CornerRadius = UDim.new(0, 2)

    local Fill = Instance.new("Frame", Track)
    Fill.Size             = UDim2.new((targetTable[key]-minVal)/(maxVal-minVal), 0, 1, 0)
    Fill.BackgroundColor3 = Color3.fromRGB(0, 180, 255)
    Fill.ZIndex           = 6
    Instance.new("UICorner", Fill).CornerRadius = UDim.new(0, 2)

    local Knob = Instance.new("Frame", Track)
    Knob.Size             = UDim2.new(0, 10, 0, 10)
    Knob.AnchorPoint      = Vector2.new(0.5, 0.5)
    Knob.Position         = UDim2.new((targetTable[key]-minVal)/(maxVal-minVal), 0, 0.5, 0)
    Knob.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    Knob.ZIndex           = 7
    Instance.new("UICorner", Knob).CornerRadius = UDim.new(0, 5)

    local function Set(ip)
        local pct = math.clamp((ip.X - Track.AbsolutePosition.X) / Track.AbsoluteSize.X, 0, 1)
        local val = minVal + pct * (maxVal - minVal)
        val = maxVal <= 10 and (math.floor(val*10)/10) or math.floor(val)
        targetTable[key]  = val
        LT.Text           = label .. ":  " .. val
        Fill.Size         = UDim2.new(pct, 0, 1, 0)
        Knob.Position     = UDim2.new(pct, 0, 0.5, 0)
        Fill.BackgroundColor3 = currentAutoColor
    end
    local drag = false
    Bg.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
            drag = true; Set(i.Position)
        end
    end)
    Bg.InputEnded:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
            drag = false
        end
    end)
    UserInputService.InputChanged:Connect(function(i)
        if drag and (i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch) then
            Set(i.Position)
        end
    end)
end
